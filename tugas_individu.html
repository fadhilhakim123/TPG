1. HIERARCHICAL CLUSTERING DENGAN DTW DISTANCE

Normalisasi Data Cabai dan Hitung Rata-Rata Harga Cabai per Provinsi

cabai_norm <- cabai_interp %>%
  filter(Produk %in% c("Cabai Merah Keriting", "Cabai Rawit Merah")) %>%
  group_by(Produk) %>%
  mutate(Harga_scaled = scale(Harga_filled)) %>%
  ungroup()

cabai_avg <- cabai_norm %>%
  group_by(NamaProv, Date) %>%
  summarize(HargaAvg = mean(Harga_scaled, na.rm = TRUE)) %>%

Bentuk Data Menjadi Wide Format

ts_wide <- cabai_avg %>%
  arrange(NamaProv, Date) %>%
  pivot_wider(
    names_from = Date,
    values_from = HargaAvg
  )

Bentuk Data Menjadi Matriks Time Series

ts_mat <- as.matrix(ts_wide[, -1])

Bentuk Clustering Menggunakan DTW Distance

dtw_dist <- proxy::dist(ts_mat, method = function(a, b) {
  dtw(a, b, distance.only = TRUE)$distance
})
hc <- hclust(dtw_dist, method = "ward.D2")
clusters <- cutree(hc, k = 4)

Visualisasi Hasil Clustering Menggunakan PCA

Bentuk Data PCA

pca <- prcomp(ts_mat, scale. = F)

pca_df <- data.frame(
  Provinsi <- ts_wide$NamaProv,
  PC1 = pca$x[,1],
  PC2 = pca$x[,2],
  Cluster = factor(clusters)
)

Visualisasi PCA dengan ggplot2

var_explained <- (pca$sdev^2) / sum(pca$sdev^2)
pc1_var <- round(var_explained[1] * 100, 1)
pc2_var <- round(var_explained[2] * 100, 1)

ggplot(pca_df, aes(PC1, PC2, color = Cluster, label = Provinsi)) +
  geom_point(size = 4) +
  geom_text(vjust = -0.8, size = 3) +
  theme_minimal() +
  labs(
    x = paste0("PC1 (", pc1_var, "%)"),
    y = paste0("PC2 (", pc2_var, "%)"),
    title = "PCA  Clustering Provinsi Berdasarkan Harga Cabai (DTW)"
  )

2. HIERARCHICAL CLUSTERING DENGAN EUCLIDEAN DISTANCE

Ekstraksi Fitur Statistik dari Data Harga Cabai

features <- cabai_interp %>%
  group_by(NamaProv) %>%
  summarise(
    mean_harga = mean(Harga, na.rm = TRUE),
    sd_harga   = sd(Harga, na.rm = TRUE),
    trend      = coef(lm(Harga ~ as.numeric(Date)))[2],   # slope
    .groups = "drop"
  )
features_no_cluster <- features[, c("NamaProv", "mean_harga", "sd_harga", "trend")]
features_scaled <- scale(features_no_cluster[, -1]) 
provinsi <- features_no_cluster$NamaProv

Bentuk Clustering dengan Euclidean Distance

# 1. Hitung jarak antar provinsi
dist_matrix <- dist(features_scaled, method = "euclidean")

# 2. Lakukan hierarchical clustering (Ward's method)
hc <- hclust(dist_matrix, method = "ward.D2")
clusters_hc <- cutree(hc, k = 4)

Visualisasi Hasil Clustering dengan PCA

Bentuk Data PCA

pca <- prcomp(features_scaled, scale. = FALSE)
